Arduino Uno SSD1306 OLED TIP120 Driver Watchdog-Protected
What this does: reads soil moisture every 10 seconds, and if the soil is dry (below 45%), runs the pump for a 12-second pulse, then waits 15 seconds before checking again. A 5-cycle safety lockout stops the system from pulsing endlessly if moisture never rises (e.g. empty reservoir), and an onboard watchdog timer auto-recovers the board if it ever hangs. Status, live countdowns, and moisture % are shown on a 128×64 OLED display.
terra_drip.ino Arduino / C++ 
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <avr/wdt.h>

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET   -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

#define SOIL_PIN A0
#define PUMP_PIN 2

int DRY_VALUE = 1023;
int WET_VALUE = 600;

const int MOISTURE_ON_THRESHOLD   = 45;
const unsigned long PUMP_RUN_TIME  = 12000;
const unsigned long CHECK_INTERVAL = 10000;
const unsigned long SOAK_TIME      = 15000;  // 15s cooldown between pulses
const int MAX_CONSECUTIVE_CYCLES   = 5;      // 5 pulses before lockout
const unsigned long DISPLAY_QUIET_AFTER_SWITCH = 800;

bool pumpState = false;
unsigned long pumpStartTime = 0;
unsigned long pumpOffTime = 0;
unsigned long lastCheck = 0;
unsigned long lastSwitchTime = 0;
const unsigned long READ_INTERVAL = 2000;

int consecutiveCycles = 0;
bool lockedOut = false;

int readMoisturePercent() {
  long total = 0;
  const int samples = 5;
  for (int i = 0; i < samples; i++) {
    total += analogRead(SOIL_PIN);
    delay(5);
  }
  int raw = total / samples;
  int percent = map(raw, DRY_VALUE, WET_VALUE, 0, 100);
  percent = constrain(percent, 0, 100);
  return percent;
}

void updateDisplay(int moisture) {
  if (millis() - lastSwitchTime < DISPLAY_QUIET_AFTER_SWITCH) return;

  unsigned long now = millis();

  display.clearDisplay();
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("TERRA-DRIP");
  display.drawLine(0, 10, 128, 10, SSD1306_WHITE);

  display.setTextSize(2);
  display.setCursor(0, 20);
  display.print("Moist:");
  display.print(moisture);
  display.println("%");

  // Status/countdown line
  display.setTextSize(1);
  display.setCursor(0, 40);
  if (lockedOut) {
    display.println("Check pump/water!");
  } else if (pumpState) {
    unsigned long elapsed = now - pumpStartTime;
    unsigned long left = (elapsed < PUMP_RUN_TIME) ? (PUMP_RUN_TIME - elapsed) / 1000 : 0;
    display.print("Watering: ");
    display.print(left);
    display.println("s left");
  } else {
    unsigned long soakElapsed = now - pumpOffTime;
    if (soakElapsed < SOAK_TIME) {
      unsigned long left = (SOAK_TIME - soakElapsed) / 1000;
      display.print("Next in: ");
      display.print(left);
      display.println("s");
    } else {
      display.println("Ready");
    }
  }

  // Pump on/off line
  display.setCursor(0, 52);
  if (lockedOut) {
    display.println("Pump: LOCKED");
  } else {
    display.print("Pump: ");
    display.println(pumpState ? "ON" : "OFF");
  }

  display.display();
}

void setup() {
  wdt_disable();
  Serial.begin(9600);
  pinMode(PUMP_PIN, OUTPUT);
  digitalWrite(PUMP_PIN, HIGH);   // HIGH = pump OFF

  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED init failed");
    while (true) { delay(1000); }
  }
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.display();

  wdt_enable(WDTO_2S);
}

void loop() {
  wdt_reset();
  unsigned long now = millis();

  if (pumpState && (now - pumpStartTime >= PUMP_RUN_TIME)) {
    pumpState = false;
    pumpOffTime = now;
    lastSwitchTime = now;
    digitalWrite(PUMP_PIN, HIGH);
    Serial.println("Pump pulse complete - OFF");
  }

  if (now - lastCheck >= CHECK_INTERVAL) {
    lastCheck = now;
    int moisture = readMoisturePercent();

    Serial.print("Moisture: "); Serial.print(moisture);
    Serial.print("%  Pump: "); Serial.println(pumpState ? "ON" : "OFF");

    bool soakOver = (now - pumpOffTime >= SOAK_TIME);

    if (!pumpState && !lockedOut && soakOver && moisture < MOISTURE_ON_THRESHOLD) {
      pumpState = true;
      pumpStartTime = now;
      lastSwitchTime = now;
      digitalWrite(PUMP_PIN, LOW);
      consecutiveCycles++;
      Serial.println("Soil dry - starting pump pulse");

      if (consecutiveCycles >= MAX_CONSECUTIVE_CYCLES) {
        lockedOut = true;
        pumpState = false;
        digitalWrite(PUMP_PIN, HIGH);
        Serial.println("SAFETY LOCKOUT: moisture not rising after several pulses");
      }
    }

    if (moisture >= MOISTURE_ON_THRESHOLD) {
      consecutiveCycles = 0;
      lockedOut = false;
    }
  }

  static unsigned long lastDisplay = 0;
  if (now - lastDisplay >= 1000) {
    lastDisplay = now;
    updateDisplay(readMoisturePercent());
  }
}
